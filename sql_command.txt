CREATE TABLE student_info (
    stu_address VARCHAR(255),
    stu_admission_date DATE NOT NULL,
    stu_admission_grade VARCHAR(20) NOT NULL,
    stu_avatar_path VARCHAR(255),
    stu_birthdate DATE NOT NULL,
    stu_chn_name VARCHAR(255) NOT NULL,
    stu_class VARCHAR(20) NOT NULL,
    stu_campus VARCHAR(50),
    stu_dormitory VARCHAR(50),
    stu_dormitory_bed VARCHAR(50),
    stu_dormitory_room VARCHAR(50),
    stu_education_length INT(8),
    stu_education_level VARCHAR(50),
    stu_email VARCHAR(255),
    stu_eng_name VARCHAR(255) NOT NULL,
    stu_ethnicity VARCHAR(50),
    stu_gender VARCHAR(10) NOT NULL,
    stu_graduation_certificate VARCHAR(255),
    stu_graduation_date DATE,
    stu_graduation_thesis_title VARCHAR(255),
    stu_id VARCHAR(32) PRIMARY KEY,
    stu_major VARCHAR(255) NOT NULL,
    stu_native_place VARCHAR(255),
    stu_nickname VARCHAR(50),
    stu_note TEXT,
    stu_prc_id VARCHAR(18) NOT NULL,
    stu_status VARCHAR(50) NOT NULL,
    stu_status_change VARCHAR(255),
    stu_study_form VARCHAR(50),
    stu_telephone VARCHAR(20)
);
